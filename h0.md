### Tee esimerkkiohjelma Pythonilla, ruutukaappaus + source code

	def vertailu(x, y):
	    if x > y: 
	        print(f"{x} on isompi kuin {y}")
	    elif x < y: 
	        print(f"{x} on pienempi kuin {y}")
	    else:
	        print(f"{x} on yhtä suuri kuin {y}")
	
	def main():
	    x = 1
	    y = 8
	    vertailu(x, y)
	
	main()
